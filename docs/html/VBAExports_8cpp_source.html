32inline int compute_all(bool isCall,
33 double S0, double r, double sigma, double T,
34 int numSteps, int numPaths, int batchSize, int seed,
35 double z,
36 double* out10)
37{
38 if (!out10) return -1;
39 if (!(S0 > 0.0) || !(sigma > 0.0) || !(T > 0.0) || numSteps <= 0 || numPaths <= 0) {
40 fill_nan(out10);
41 return -2;
42 }
43
44 try {
45 BlackScholesModel model(r, sigma, S0);
46 MonteCarloSimulator mc(model, numPaths, numSteps, batchSize, seed);
47 GreeksCalculator gc;
48
49 if (isCall) {
50 LookbackFloatingCallPayoff payoff;
51 LookbackCallOption option(payoff, T);
52 const Greeks g = gc.computeGreeks_t0(option, mc, /*t0=*/0.0, z);
53 fill_out10(g, out10);
54 } else {
55 LookbackFloatingPutPayoff payoff;
56 LookbackPutOption option(payoff, T);
57 const Greeks g = gc.computeGreeks_t0(option, mc, /*t0=*/0.0, z);
58 fill_out10(g, out10);
59 }
60 return 0;
61 } catch (...) {
62 fill_nan(out10);
63 return -99;
64 }
65}

67inline int compute_curve(bool isCall,
68 double spotStart, double spotStep, int nPoints,
69 double r, double sigma, double T,
70 int numSteps, int numPaths, int batchSize, int seed,
71 double z,
72 double* spots, double* prices, double* deltas)
73{
74 if (!spots || !prices || !deltas) return -1;
75 if (nPoints <= 0) return -2;
76 if (!(spotStep > 0.0) || !(spotStart > 0.0) || !(sigma > 0.0) || !(T > 0.0)) return -3;
77
78 try {
79 // IMPORTANT:
80 // For the charts we only need PRICE and DELTA, not all Greeks.
81 // Calling computeGreeks_t0 here would run many extra MC pricers per point
82 // (delta/gamma/vega/rho/theta bumps). This version is ~8x faster.
83 //
84 // For floating-strike lookbacks in Black–Scholes, P(S0) is homogeneous of degree 1:
85 // P(S0) = S0 * C => Delta = dP/dS0 = C = P(S0)/S0, Gamma ≈ 0.
86 // So we compute delta from the price estimate.
87
88 if (isCall) {
89 LookbackFloatingCallPayoff payoff;
90 LookbackCallOption option(payoff, T);
91
92 for (int i = 0; i < nPoints; ++i) {
93 const double S0 = spotStart + static_cast<double>(i) * spotStep;
94 spots[i] = S0;
95
96 BlackScholesModel model(r, sigma, S0);
97
98 // Use a different seed per point to avoid perfectly "rigid" straight lines
99 // when reusing the exact same random numbers for every S0.
100 const int seed_i = seed + i * 10007;
101
102 MonteCarloSimulator mc(model, numPaths, numSteps, batchSize, seed_i);
103 const auto res = mc.simulateOptionPriceControlVariate_t0(option, /*t0=*/0.0, z);
104
105 prices[i] = res.price;
106 deltas[i] = (S0 > 0.0) ? (res.price / S0) : 0.0;
107 }
108 } else {
109 LookbackFloatingPutPayoff payoff;
110 LookbackPutOption option(payoff, T);
111
112 for (int i = 0; i < nPoints; ++i) {
113 const double S0 = spotStart + static_cast<double>(i) * spotStep;
114 spots[i] = S0;
115
116 BlackScholesModel model(r, sigma, S0);
117 const int seed_i = seed + i * 10007;
118
119 MonteCarloSimulator mc(model, numPaths, numSteps, batchSize, seed_i);
120 const auto res = mc.simulateOptionPriceControlVariate_t0(option, /*t0=*/0.0, z);
121
122 prices[i] = res.price;
123 deltas[i] = (S0 > 0.0) ? (res.price / S0) : 0.0;
124 }
125 }
126
127 return 0;
128 } catch (...) {
129 const double nan = std::numeric_limits<double>::quiet_NaN();
130 for (int i = 0; i < nPoints; ++i) {
131 spots[i] = nan;
132 prices[i] = nan;
133 deltas[i] = nan;
134 }
135 return -99;
136 }
137}
