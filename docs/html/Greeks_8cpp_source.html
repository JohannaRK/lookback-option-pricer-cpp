32 const double hT = epsT_;

34 Greeks g;

36 // -------------------------------------------------------------------------
37 // PRICE (Control Variate) + (optional CI available in MCResult)
38 // We store only the point estimate in Greeks.
39 // -------------------------------------------------------------------------
40 const auto baseRes = mcBase.simulateOptionPriceControlVariate_t0(option, t0, z);
41 g.price = baseRes.price;
42
43 // Store MC uncertainty for the price only
44 g.priceStdError = baseRes.stdError;
45 g.priceCILower = baseRes.ciLower;
46 g.priceCIUpper = baseRes.ciUpper;

48 // DELTA & GAMMA
49 // -------------------------------------------------------------------------
50 {
51 BlackScholesModel mUp = model; mUp.setInitialPrice(S0 + hS);
52 BlackScholesModel mDn = model; mDn.setInitialPrice(S0 - hS);
53
54 MonteCarloSimulator mcUp(mUp, mcBase.getNumPaths(), mcBase.getNumSteps(),
55 mcBase.getBatchSize(), mcBase.getSeed());

57 mcBase.getBatchSize(), mcBase.getSeed());
58

60 const double pDn = mcDn.simulateOptionPriceControlVariate_t0(option, t0, z).price;
61

64 }

66 // -------------------------------------------------------------------------
67 // VEGA
68 // -------------------------------------------------------------------------

86 // -------------------------------------------------------------------------

90

92 mcBase.getBatchSize(), mcBase.getSeed());
93 MonteCarloSimulator mcDn(mDn, mcBase.getNumPaths(), mcBase.getNumSteps(),
94 mcBase.getBatchSize(), mcBase.getSeed());
95
96 const double pUp = mcUp.simulateOptionPriceControlVariate_t0(option, t0, z).price;
97 const double pDn = mcDn.simulateOptionPriceControlVariate_t0(option, t0, z).price;
98
99 g.rho = (pUp - pDn) / (2.0 * hR);
100 }
101
102 // -------------------------------------------------------------------------
103 // THETA (one-sided, time-decay / current-time bump)
104 // We want theta = ∂P/∂t. Since τ = T - t, we have ∂P/∂t = -∂P/∂τ.
105 // Approximating a decrease of τ by hT is equivalent to reducing maturity T by hT
106 // while keeping t0 fixed.
107 // -------------------------------------------------------------------------
108 {
109 const double Tdn = std::max(T - hT, 1e-6);
110 auto optDn = option.cloneWithMaturity(Tdn);
111
112 MonteCarloSimulator mcSame(model, mcBase.getNumPaths(), mcBase.getNumSteps(),
113 mcBase.getBatchSize(), mcBase.getSeed());
114
115 const double pDn = mcSame.simulateOptionPriceControlVariate_t0(*optDn, t0, z).price;
116
117 // theta = ∂P/∂t at t0 (time decay). With maturity fixed, increasing t reduces
118 // time-to-maturity. Using a one-sided bump on time-to-maturity:
119 // theta ≈ (P(T - hT) - P(T)) / hT
120 g.theta = (pDn - g.price) / hT;
121 }
122
123 return g;
124}
